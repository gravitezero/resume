\newcommand{\mycolor}[1]{
	\textcolor[RGB]{59,134,254}{#1}
}

\newcommand{\rmp}[1]{
	\begin{minipage}[t]{\rigthCol}
		\begin{flushright}
			#1
		\end{flushright}
	\end{minipage}	
}

\newcommand{\lmp}[1]{
	\begin{minipage}[t]{\leftCol}
		\begin{flushleft}
			#1
		\end{flushleft}
	\end{minipage}
}

\newcommand{\head}[1]{
	\begin{minipage}[t]{\titleCol}
		\begin{flushleft}
			#1
		\end{flushleft}
	\end{minipage}
}

\newcommand{\rltabvline}[3]{
	\begin{tabular}{@{} p{\rigthCol} @{\hspace{\vSpaceMargin}} | @{\hspace{\vSpaceMargin}} p{\titleCol} @{} p{\titleCol}}
	\rmp{#1} & \head{#2} & \head{#3}
	\end{tabular}
}		

\newcommand{\rltab}[2]{
	\begin{tabular}{@{} p{\rigthCol} @{\hspace{2\vSpaceMargin}} p{\leftCol} @{}}
	\rmp{#1} & \lmp{#2}
	\end{tabular}
}

\newcommand{\paragraphTitle}[1]{
		\huge{\mycolor{#1}}
}

\newcommand{\header}[3]{
	\rltabvline{#1}{\LARGE{#2}}{\LARGE{#3}}
	\vspace{\bottomMarginHeader}\\
}

\renewcommand{\section}[1]{
	\vspace{\topMarginSection}
	\rltab{}{\paragraphTitle{#1}}
	\vspace{\bottomMarginSection}\\
}

\renewcommand{\subsection}[2]{
	\rltab{\textbf{#1}}{#2}\\
}
